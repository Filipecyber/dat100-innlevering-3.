package no.hvl.dat100.oppgave4;

import java.io.File;
import java.io.FileNotFoundException;
import java.io.PrintWriter;

import no.hvl.dat100.oppgave3.Blogg;

public class SkrivBlogg {

    public static boolean skriv(Blogg samling, String mappe, String filnavn) {
        PrintWriter pw = null;
        try {
            // Bygg filsti robust med File for riktig separator på alle plattformer
            File dir = (mappe == null || mappe.isEmpty()) ? new File(".") : new File(mappe);
            File file = new File(dir, filnavn);

            pw = new PrintWriter(file);
            
            pw.print(samling.toString());
            return true;
        } catch (FileNotFoundException e) {
            // Returnerer false ved skrivefeil 
            return false;
        } finally {
            // Sørger for at ressursen lukkes selv ved unntak
            if (pw != null) {
                pw.close();
            }
        }
    }
}
